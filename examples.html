Примеры fetch-запросов

Эти примеры можно вставить в любой простой frontend-проект.

Панель Витрина Документация Код

Базовая функция

const API_BASE = "https://front.bilimai.kg/api";
const TOKEN = "your_token";

async function apiRequest(path, { method = "GET", body } = {}) {
  const url = new URL(API_BASE + path);
  url.searchParams.set("token", TOKEN);

  const response = await fetch(url, {
    method,
    headers: {
      "Accept": "application/json",
      "Content-Type": "application/json"
    },
    body: body ? JSON.stringify(body) : undefined
  });

  const data = await response.json().catch(() => null);

  if (!response.ok) {
    const details = data?.errors
      ? Object.entries(data.errors).map(([field, messages]) => `${field}: ${messages.join(", ")}`).join("\n")
      : "";

    throw new Error([data?.message || data?.error || `HTTP ${response.status}`, details].filter(Boolean).join("\n"));
  }

  return data;
}
